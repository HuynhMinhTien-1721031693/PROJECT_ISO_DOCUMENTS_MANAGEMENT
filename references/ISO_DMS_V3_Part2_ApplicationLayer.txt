1. Tổng quan Application Layer
Application Layer orchestrate use cases. Mỗi use case là một Command (write) hoặc Query (read), xử lý bởi một Handler thông qua MediatR pipeline.

Nguyên tắc thiết kế:
•	Mỗi folder = một use case: Command/Query + Handler + Validator trong cùng thư mục
•	Handler không chứa business logic — chỉ gọi Domain methods và Infrastructure services
•	Pipeline Behaviours xử lý cross-cutting: logging, validation, authorization
•	Không reference EF Core, Azure SDK, hoặc bất kỳ HTTP concern nào

2. MediatR Pipeline Behaviours
3 Behaviours đăng ký theo thứ tự: LoggingBehaviour → ValidationBehaviour → AuthorizationBehaviour → Handler. Thứ tự quan trọng: log trước, validate sau, authorize cuối cùng trước khi handler chạy.

2.1  Thứ tự thực thi và vai trò
Behaviour	Thứ tự	Vai trò
LoggingBehaviour	1 (đầu tiên)	Ghi log request name, user, duration. Warn nếu >500ms
ValidationBehaviour	2	Chạy tất cả IValidator<T> đã đăng ký. Throw nếu có lỗi
AuthorizationBehaviour	3	Kiểm tra [Authorize] attribute và RBAC permission
Handler (actual)	4 (cuối)	Business use case logic

2.2  ValidationBehaviour.cs
Validators chạy song song (Task.WhenAll) để tối ưu hiệu suất. Tất cả lỗi được thu thập trước khi throw — client nhận đủ danh sách lỗi trong một lần.

using FluentValidation;
using MediatR;
using Microsoft.Extensions.Logging;

namespace IsoDoc.Application.Common.Behaviours;

/// <summary>
/// MediatR pipeline behaviour that runs FluentValidation validators
/// BEFORE the handler executes.
///
/// If any validator is registered for the request type, all validators run.
/// On failure: throws ValidationException (caught by ExceptionHandlingMiddleware
/// in WebAPI and mapped to HTTP 400 Problem Details).
///
/// No validators registered → request passes through untouched.
/// </summary>
public sealed class ValidationBehaviour<TRequest, TResponse>
    : IPipelineBehavior<TRequest, TResponse>
    where TRequest : notnull
{
    private readonly IEnumerable<IValidator<TRequest>> _validators;
    private readonly ILogger<ValidationBehaviour<TRequest, TResponse>> _logger;

    public ValidationBehaviour(
        IEnumerable<IValidator<TRequest>> validators,
        ILogger<ValidationBehaviour<TRequest, TResponse>> logger)
    {
        _validators = validators;
        _logger     = logger;
    }

    public async Task<TResponse> Handle(
        TRequest                          request,
        RequestHandlerDelegate<TResponse> next,
        CancellationToken                 cancellationToken)
    {
        if (!_validators.Any())
            return await next();

        var context = new ValidationContext<TRequest>(request);

        // Run all validators in parallel
        var validationResults = await Task.WhenAll(
            _validators.Select(v => v.ValidateAsync(context, cancellationToken)));

        var failures = validationResults
            .SelectMany(r => r.Errors)
            .Where(f => f is not null)
            .ToList();

        if (failures.Any())
        {
            _logger.LogWarning(
                "Validation failed for {RequestType}. Errors: {Errors}",
                typeof(TRequest).Name,
                string.Join("; ", failures.Select(f => $"{f.PropertyName}: {f.ErrorMessage}")));

            throw new ValidationException(failures);
        }

        return await next();
    }
}


2.3  LoggingBehaviour.cs + AuthorizationBehaviour.cs
using IsoDoc.Application.Common.Interfaces;
using MediatR;
using Microsoft.Extensions.Logging;
using System.Diagnostics;

namespace IsoDoc.Application.Common.Behaviours;

/// <summary>
/// Logs every request: name, user, duration, and whether it succeeded.
/// Slow requests (>500ms) are logged as warnings — helps identify N+1 queries.
/// </summary>
public sealed class LoggingBehaviour<TRequest, TResponse>
    : IPipelineBehavior<TRequest, TResponse>
    where TRequest : notnull
{
    private const int SlowRequestThresholdMs = 500;

    private readonly ILogger<LoggingBehaviour<TRequest, TResponse>> _logger;
    private readonly ICurrentUserService _currentUser;

    public LoggingBehaviour(
        ILogger<LoggingBehaviour<TRequest, TResponse>> logger,
        ICurrentUserService currentUser)
    {
        _logger      = logger;
        _currentUser = currentUser;
    }

    public async Task<TResponse> Handle(
        TRequest                          request,
        RequestHandlerDelegate<TResponse> next,
        CancellationToken                 cancellationToken)
    {
        var requestName = typeof(TRequest).Name;
        var userId      = _currentUser.UserId?.ToString() ?? "anonymous";

        _logger.LogInformation(
            "Handling {RequestName} | User: {UserId}",
            requestName, userId);

        var sw = Stopwatch.StartNew();
        try
        {
            var response = await next();
            sw.Stop();

            if (sw.ElapsedMilliseconds > SlowRequestThresholdMs)
            {
                _logger.LogWarning(
                    "SLOW REQUEST: {RequestName} took {ElapsedMs}ms | User: {UserId}",
                    requestName, sw.ElapsedMilliseconds, userId);
            }
            else
            {
                _logger.LogInformation(
                    "Handled {RequestName} in {ElapsedMs}ms",
                    requestName, sw.ElapsedMilliseconds);
            }

            return response;
        }
        catch (Exception ex)
        {
            sw.Stop();
            _logger.LogError(ex,
                "Error handling {RequestName} after {ElapsedMs}ms | User: {UserId}",
                requestName, sw.ElapsedMilliseconds, userId);
            throw;
        }
    }
}

/// <summary>
/// Checks RBAC permissions before the handler runs.
/// Requests decorated with [Authorize(Permission = "...")] are checked
/// against the current user's role-permission set.
///
/// Unannotated requests pass through (public queries like search).
/// </summary>
public sealed class AuthorizationBehaviour<TRequest, TResponse>
    : IPipelineBehavior<TRequest, TResponse>
    where TRequest : notnull
{
    private readonly ICurrentUserService    _currentUser;
    private readonly IPermissionService     _permissions;
    private readonly ILogger<AuthorizationBehaviour<TRequest, TResponse>> _logger;

    public AuthorizationBehaviour(
        ICurrentUserService    currentUser,
        IPermissionService     permissions,
        ILogger<AuthorizationBehaviour<TRequest, TResponse>> logger)
    {
        _currentUser = currentUser;
        _permissions = permissions;
        _logger      = logger;
    }

    public async Task<TResponse> Handle(
        TRequest                          request,
        RequestHandlerDelegate<TResponse> next,
        CancellationToken                 cancellationToken)
    {
        // Check for [Authorize] attribute on the request class
        var authorizeAttrs = typeof(TRequest)
            .GetCustomAttributes(typeof(AuthorizeAttribute), inherit: true)
            .Cast<AuthorizeAttribute>()
            .ToList();

        if (!authorizeAttrs.Any())
            return await next(); // No auth required

        if (_currentUser.UserId is null)
            throw new UnauthorizedAccessException("User is not authenticated.");

        foreach (var attr in authorizeAttrs.Where(a => !string.IsNullOrEmpty(a.Permission)))
        {
            var hasPermission = await _permissions.HasPermissionAsync(
                _currentUser.UserId.Value, attr.Permission!, cancellationToken);

            if (!hasPermission)
            {
                _logger.LogWarning(
                    "Authorization failed: User {UserId} missing permission '{Permission}' for {Request}",
                    _currentUser.UserId, attr.Permission, typeof(TRequest).Name);

                throw new ForbiddenAccessException(
                    _currentUser.UserId.Value, attr.Permission!);
            }
        }

        return await next();
    }
}

{

    public string Permission { get; }

    public ForbiddenAccessException(Guid userId, string permission)
        : base($"User '{userId}' does not have permission '{permission}'.")
    {
        UserId     = userId;
        Permission = permission;
    }
}

 
3. Common Models — Result<T> và DTOs
Result<T> là envelope chứa kết quả trả về từ Handler. Thay vì throw exception cho business errors, Handler trả về Result.Failure(). Exception chỉ throw cho lỗi ngoài kiểm soát (DB down, network fail).

3.1  Result<T> + PagedList<T>
namespace IsoDoc.Application.Common.Models;

/// <summary>
/// Generic result wrapper. All command/query handlers return this
/// instead of throwing exceptions — callers decide how to handle errors.
/// </summary>
public class Result<T>
{
    public bool     IsSuccess  { get; }
    public T?       Value      { get; }
    public string?  Error      { get; }
    public string?  ErrorCode  { get; }

    private Result(bool isSuccess, T? value, string? error, string? errorCode)
    {
        IsSuccess = isSuccess;
        Value     = value;
        Error     = error;
        ErrorCode = errorCode;
    }

    public static Result<T> Success(T value)
        => new(true, value, null, null);

    public static Result<T> Failure(string error, string errorCode = "GENERAL_ERROR")
        => new(false, default, error, errorCode);

    public static implicit operator Result<T>(T value) => Success(value);
}

/// <summary>Non-generic result for commands that return no value.</summary>
public class Result
{
    public bool    IsSuccess { get; }
    public string? Error     { get; }
    public string? ErrorCode { get; }

    private Result(bool isSuccess, string? error, string? errorCode)
    {
        IsSuccess = isSuccess;
        Error     = error;
        ErrorCode = errorCode;
    }

    public static Result Success()
        => new(true, null, null);

    public static Result Failure(string error, string errorCode = "GENERAL_ERROR")
        => new(false, error, errorCode);
}

/// <summary>
/// Standard paged list for query results. 
/// Used by SearchDocumentsQuery, GetPendingApprovalsQuery, etc.
/// </summary>
public class PagedList<T>
{
    public IReadOnlyList<T> Items     { get; }
    public int              Page      { get; }
    public int              PageSize  { get; }
    public int              TotalCount { get; }
    public int              TotalPages => (int)Math.Ceiling(TotalCount / (double)PageSize);
    public bool             HasNextPage => Page < TotalPages;
    public bool             HasPreviousPage => Page > 1;

    public PagedList(IReadOnlyList<T> items, int totalCount, int page, int pageSize)
    {
        Items      = items;
        TotalCount = totalCount;
        Page       = page;
        PageSize   = pageSize;
    }
}

// ── Validator ────────────────────────────────────────────────────────────────

public sealed class UploadDocumentCommandValidator : AbstractValidator<UploadDocumentCommand>
{
    private static readonly string[] AllowedContentTypes =
        ["application/pdf", "application/vnd.openxmlformats-officedocument.wordprocessingml.document",
         "application/vnd.openxmlformats-officedocument.spreadsheetml.sheet"];

    private const long MaxFileSizeBytes = 50 * 1024 * 1024; // 50 MB

    public UploadDocumentCommandValidator()
    {
        RuleFor(x => x.Title)
            .NotEmpty().WithMessage("Tiêu đề tài liệu không được để trống.")
            .MaximumLength(500).WithMessage("Tiêu đề không quá 500 ký tự.");

        RuleFor(x => x.DocumentCode)
            .NotEmpty().WithMessage("Mã tài liệu không được để trống.")
            .Matches(@"^[A-Z]{2,6}-[A-Z]{2,4}-\d{3}$")
            .WithMessage("Mã tài liệu phải theo định dạng: PREFIX-TYPE-NNN (ví dụ: QMS-PR-001).");

        RuleFor(x => x.Standard)
            .IsInEnum().WithMessage("Tiêu chuẩn ISO không hợp lệ.");

        RuleFor(x => x.Category)
            .IsInEnum().WithMessage("Phân loại tài liệu không hợp lệ.");

        RuleFor(x => x.FileName)
            .NotEmpty().WithMessage("Tên file không được để trống.");

        RuleFor(x => x.ContentType)
            .Must(ct => AllowedContentTypes.Contains(ct))
            .WithMessage("Chỉ chấp nhận file PDF, DOCX hoặc XLSX.");

        RuleFor(x => x.FileSize)
            .GreaterThan(0).WithMessage("File không được rỗng.")
            .LessThanOrEqualTo(MaxFileSizeBytes)
            .WithMessage($"Dung lượng file không vượt quá {MaxFileSizeBytes / 1024 / 1024} MB.");

        RuleFor(x => x.ChecksumHex)
            .NotEmpty().WithMessage("Checksum SHA-256 bắt buộc phải có.")
            .Length(64).WithMessage("Checksum phải là chuỗi hex 64 ký tự (SHA-256).");

        RuleFor(x => x.Tags)
            .Must(tags => tags.Count <= 10).WithMessage("Tối đa 10 tags.")
            .ForEach(tag => tag.MaximumLength(50).WithMessage("Mỗi tag tối đa 50 ký tự."));
    }
}

// ── Handler ───────────────────────────────────────────────────────────────────

public sealed class UploadDocumentCommandHandler
    : IRequestHandler<UploadDocumentCommand, Result<Guid>>
{
    private readonly IDocumentRepository  _documents;
    private readonly IFileStorageService  _fileStorage;
    private readonly ICurrentUserService  _currentUser;
    private readonly IAuditService        _audit;
    private readonly ILogger<UploadDocumentCommandHandler> _logger;

    public UploadDocumentCommandHandler(
        IDocumentRepository  documents,
        IFileStorageService  fileStorage,
        ICurrentUserService  currentUser,
        IAuditService        audit,
        ILogger<UploadDocumentCommandHandler> logger)
    {
        _documents   = documents;
        _fileStorage = fileStorage;
        _currentUser = currentUser;
        _audit       = audit;
        _logger      = logger;
    }

    public async Task<Result<Guid>> Handle(
        UploadDocumentCommand command,
        CancellationToken     ct)
    {
        // 1. Check for duplicate document code
        if (await _documents.ExistsAsync(command.DocumentCode, ct))
            return Result<Guid>.Failure(
                $"Mã tài liệu '{command.DocumentCode}' đã tồn tại trong hệ thống.",
                "DOCUMENT_CODE_DUPLICATE");

        var ownerId = _currentUser.UserId!.Value;

        // 2. Create domain entity (raises DocumentCreatedEvent)
        var document = Document.Create(
            title:        command.Title,
            code:         command.DocumentCode,
            standard:     command.Standard,
            category:     command.Category,
            ownerId:      ownerId,
            departmentId: Guid.Empty, // TODO: resolve from user's department
            description:  command.Description,
            tags:         command.Tags);

        // 3. Upload file to Blob Storage
        string blobPath;
        try
        {
            blobPath = await _fileStorage.UploadAsync(
                fileStream:  command.FileStream,
                fileName:    command.FileName,
                contentType: command.ContentType,
                documentId:  document.Id,
                ct:          ct);
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "File upload failed for document {Code}", command.DocumentCode);
            return Result<Guid>.Failure(
                "Lỗi khi tải file lên. Vui lòng thử lại.",
                "FILE_UPLOAD_FAILED");
        }

        // 4. Add version to aggregate
        var fileType = command.ContentType switch
        {
            "application/pdf" => DocumentFileType.Pdf,
            var ct2 when ct2.Contains("wordprocessingml") => DocumentFileType.Docx,
            var ct2 when ct2.Contains("spreadsheetml")    => DocumentFileType.Xlsx,
            _ => DocumentFileType.Pdf
        };

        document.AddVersion(
            blobPath:    blobPath,
            fileSize:    command.FileSize,
            fileType:    fileType,
            checksumHex: command.ChecksumHex,
            uploadedBy:  ownerId,
            changeNote:  command.ChangeNote);

        // 5. Persist
        await _documents.AddAsync(document, ct);
        await _documents.SaveChangesAsync(ct);
        // Domain events dispatched by EF Core SaveChanges interceptor

        // 6. Audit log (ISO 27001)
        await _audit.LogAsync(
            userId:     ownerId,
            action:     "Document.Upload",
            entityType: "Document",
            entityId:   document.Id.ToString(),
            newValues:  new { document.Id, document.Code, command.FileName },
            ct:         ct);

        _logger.LogInformation(
            "Document created: {Code} by user {UserId}", document.Code, ownerId);

        return Result<Guid>.Success(document.Id);
    }
}

            "Document {Code} submitted for approval by {UserId}", document.Code, userId);

        return Result<Guid>.Success(workflow.Id);

public sealed record RecordDecisionCommand : IRequest<Result>

    public WorkflowDecision Decision   { get; init; }

public sealed class RecordDecisionCommandValidator
    : AbstractValidator<RecordDecisionCommand>

    {
        RuleFor(x => x.WorkflowId)
            .NotEmpty().WithMessage("WorkflowId không được để trống.");

        RuleFor(x => x.Decision)
            .Must(d => d is WorkflowDecision.Approved or WorkflowDecision.Rejected)
            .WithMessage("Decision phải là Approved hoặc Rejected.");

        RuleFor(x => x.Comment)
            .NotEmpty().When(x => x.Decision == WorkflowDecision.Rejected)
            .WithMessage("Bắt buộc phải nhập lý do khi từ chối.");

        RuleFor(x => x.Comment)
            .MaximumLength(2000).When(x => x.Comment is not null)
            .WithMessage("Comment không quá 2000 ký tự.");
    }
}

public sealed class RecordDecisionCommandHandler
    : IRequestHandler<RecordDecisionCommand, Result>
{
    private readonly IApprovalWorkflowRepository _workflows;
    private readonly IDocumentRepository         _documents;
    private readonly ICurrentUserService         _currentUser;
    private readonly IAuditService               _audit;
    private readonly ILogger<RecordDecisionCommandHandler> _logger;

    public RecordDecisionCommandHandler(
        IApprovalWorkflowRepository workflows,
        IDocumentRepository         documents,
        ICurrentUserService         currentUser,
        IAuditService               audit,
        ILogger<RecordDecisionCommandHandler> logger)
    {
        _workflows   = workflows;
        _documents   = documents;
        _currentUser = currentUser;
        _audit       = audit;
        _logger      = logger;
    }

    public async Task<Result> Handle(RecordDecisionCommand command, CancellationToken ct)
    {
        var workflow = await _workflows.GetByDocumentIdAsync(command.WorkflowId, ct);
        if (workflow is null)
            return Result.Failure("Không tìm thấy workflow.", "WORKFLOW_NOT_FOUND");

        var document = await _documents.GetByIdAsync(workflow.DocumentId, ct);
        if (document is null)
            return Result.Failure("Không tìm thấy tài liệu.", "DOCUMENT_NOT_FOUND");

        var userId = _currentUser.UserId!.Value;

        // Workflow records the decision (validates approver identity internally)
        workflow.RecordDecision(userId, command.Decision, command.Comment);

        // Sync document state based on workflow outcome
        if (workflow.Status == WorkflowStatus.Approved)
        {
            document.Publish(userId, isMajorChange: false);
        }
        else if (workflow.Status == WorkflowStatus.Rejected)
        {
            document.Reject(userId, command.Comment ?? "No reason provided.");
        }
        else
        {
            // Still in progress — advance document to PendingFinalApproval
            if (workflow.CurrentStepOrder == 2)
                document.AdvanceToFinalApproval(userId);
        }

        _workflows.Update(workflow);
        _documents.Update(document);
        await _documents.SaveChangesAsync(ct);

        var action = command.Decision == WorkflowDecision.Approved
            ? "Workflow.Approved" : "Workflow.Rejected";

        await _audit.LogAsync(userId, action, "ApprovalWorkflow",
            workflow.Id.ToString(),
            newValues: new { command.Decision, command.Comment }, ct: ct);

        _logger.LogInformation(
            "Workflow {WorkflowId} step {Step}: {Decision} by {UserId}",
            workflow.Id, workflow.CurrentStepOrder, command.Decision, userId);

        return Result.Success();
    }
}

7. Domain Event Handlers
Handlers implement INotificationHandler<TEvent> và được dispatch SAU khi SaveChanges() thành công. Không nằm trong transaction — nếu handler fail, document vẫn được lưu, chỉ notification bị miss. Dùng outbox pattern cho production nếu cần đảm bảo.

7.1  Dispatch flow
// EF Core SaveChanges interceptor (Infrastructure) dispatch flow:
1. Handler calls _documents.SaveChangesAsync()  → EF commits to DB
2. SaveChangesInterceptor.SavingChangesAsync()  → collect domain events
3. After commit: foreach event → _mediator.Publish(event)
4. MediatR routes to matching INotificationHandler<T>
   → DocumentPublishedEventHandler  → update Elasticsearch + notify
   → DocumentRejectedEventHandler   → notify owner
   → WorkflowStepAdvancedHandler    → notify next approver

7.2  Event Handlers code
using IsoDoc.Application.Common.Interfaces;

namespace IsoDoc.Application.Documents.EventHandlers;

/// <summary>
/// Triggered after a document is published.
/// Runs AFTER the database transaction commits (dispatched by EF Core SaveChanges interceptor).
///
/// Responsibilities:
/// 1. Update Elasticsearch search index
/// 2. Send in-app + email notifications to all viewers
/// 3. Archive the previous published version
/// </summary>
public sealed class DocumentPublishedEventHandler
    : INotificationHandler<DocumentPublishedEvent>
{
    private readonly ISearchService       _search;
    private readonly IDocumentRepository  _documents;
    private readonly INotificationSender  _notifications;
    private readonly ILogger<DocumentPublishedEventHandler> _logger;

    public DocumentPublishedEventHandler(
        ISearchService       search,
        IDocumentRepository  documents,
        INotificationSender  notifications,
        ILogger<DocumentPublishedEventHandler> logger)
    {
        _search        = search;
        _documents     = documents;
        _notifications = notifications;
        _logger        = logger;
    }

    public async Task Handle(DocumentPublishedEvent notification, CancellationToken ct)
    {
        _logger.LogInformation(
            "Handling DocumentPublishedEvent for {Code} v{Version}",
            notification.DocumentCode, notification.Version);

        var document = await _documents.GetByIdAsync(notification.DocumentId, ct);
        if (document is null)
        {
            _logger.LogWarning("Document {Id} not found for indexing.", notification.DocumentId);
            return;
        }

        // 1. Update search index
        await _search.UpdateDocumentIndexAsync(document, ct);

        // 2. Notify (fire-and-forget with error swallow — not critical path)
        try
        {
            await _notifications.SendInAppNotificationAsync(
                userId:    notification.ApprovedBy,
                title:     "Tài liệu đã được phát hành",
                message:   $"Tài liệu {notification.DocumentCode} v{notification.Version} đã được phê duyệt và phát hành.",
                actionUrl: $"/documents/{notification.DocumentId}",
                ct:        ct);
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "Failed to send publish notification for {Code}", notification.DocumentCode);
        }
    }
}

    public async Task Handle(WorkflowStepAdvancedEvent notification, CancellationToken ct)
    {
        var document = await _documents.GetByIdAsync(notification.DocumentId, ct);
        if (document is null) return;

        await _notifications.SendInAppNotificationAsync(
            userId:    notification.NextApproverId,
            title:     $"Cần phê duyệt: {document.Code}",
            message:   $"Tài liệu '{document.Title}' đang chờ phê duyệt bước {notification.NextStepOrder} của bạn.",
            actionUrl: $"/approvals/{notification.DocumentId}",
            ct:        ct);

        _logger.LogInformation(
            "Approval request sent to {ApproverId} for document {Code} step {Step}",
            notification.NextApproverId, document.Code, notification.NextStepOrder);
    }
}

 
8. Dependency Injection — DependencyInjection.cs
Một dòng duy nhất trong Program.cs: builder.Services.AddApplicationServices() — đăng ký MediatR, FluentValidation, AutoMapper, và tất cả Pipeline Behaviours.

using FluentValidation;
using IsoDoc.Application.Common.Behaviours;
using MediatR;
using Microsoft.Extensions.DependencyInjection;
using System.Reflection;

namespace IsoDoc.Application;

/// <summary>
/// Extension method for IServiceCollection.
/// Called from WebAPI/Program.cs: services.AddApplicationServices()
///
/// Registers:
/// - MediatR (all handlers in this assembly)
/// - FluentValidation (all validators in this assembly)
/// - Pipeline behaviours (in order: Logging → Validation → Authorization)
/// - AutoMapper profiles
/// </summary>
public static class DependencyInjection
{
    public static IServiceCollection AddApplicationServices(
        this IServiceCollection services)
    {
        var assembly = Assembly.GetExecutingAssembly();

        // MediatR — scans assembly for IRequestHandler, INotificationHandler
        services.AddMediatR(cfg =>
        {
            cfg.RegisterServicesFromAssembly(assembly);

            // Pipeline behaviours execute in registration order:
            cfg.AddBehavior(typeof(IPipelineBehavior<,>), typeof(LoggingBehaviour<,>));
            cfg.AddBehavior(typeof(IPipelineBehavior<,>), typeof(ValidationBehaviour<,>));
            cfg.AddBehavior(typeof(IPipelineBehavior<,>), typeof(AuthorizationBehaviour<,>));
        });

        // FluentValidation — scans assembly for AbstractValidator<T>
        services.AddValidatorsFromAssembly(assembly);

        // AutoMapper — scans assembly for Profile subclasses
        services.AddAutoMapper(assembly);

        return services;
    }
}


8.1  Đăng ký trong Program.cs (WebAPI)
// Program.cs
var builder = WebApplication.CreateBuilder(args);

builder.Services
    .AddApplicationServices()      // ← Application layer (MediatR, Validators, AutoMapper)
    .AddInfrastructureServices()   // ← Infrastructure (EF Core, Blob, Redis, Search)
    .AddWebApiServices();          // ← Controllers, Swagger, Auth middleware

var app = builder.Build();
app.UseExceptionHandling();        // ← Map DomainException → HTTP 400/404/422
app.UseAuthentication();
app.UseAuthorization();
app.MapControllers();
app.Run();
