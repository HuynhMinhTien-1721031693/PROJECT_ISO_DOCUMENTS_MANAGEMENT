1. Tổng quan WebAPI Layer
WebAPI Layer là điểm vào duy nhất của hệ thống từ HTTP. Tầng này không chứa business logic — chỉ nhận HTTP request, validate format, dispatch sang MediatR, và format lại response.

Cấu trúc:
IsoDoc.WebAPI/
├── Program.cs                          ← Entry point, bootstrap, pipeline config
├── Extensions/
│   └── WebApiServiceExtensions.cs      ← DI: Controllers, Swagger, JWT, CORS, HealthChecks
├── Middleware/
│   └── ExceptionHandlingMiddleware.cs  ← Global catch → RFC 7807 ProblemDetails
├── Models/
│   └── ApiResponse.cs                  ← ApiResponse<T> envelope + ApiControllerBase
└── Controllers/
    ├── DocumentsController.cs          ← Documents CRUD + upload + submit
    └── OtherControllers.cs             ← WorkflowController, SearchController, AuthController
 
2. Program.cs — Entry Point
Program.cs khởi động hệ thống theo thứ tự: bootstrap logger → DI services → HTTP pipeline → run. Serilog được khởi tạo trước DI để capture lỗi startup.

Sử dụng bootstrap logger pattern: nếu DI fail khi khởi động (ví dụ connection string sai), lỗi vẫn được ghi ra console trước khi process exit.

using IsoDoc.Application;
using IsoDoc.Infrastructure;
using IsoDoc.WebAPI.Extensions;
using IsoDoc.WebAPI.Middleware;
using Microsoft.AspNetCore.Mvc;
using Serilog;

// ── Bootstrap logger (captures startup errors before DI is ready) ─────────────
Log.Logger = new LoggerConfiguration()
    .WriteTo.Console()
    .CreateBootstrapLogger();

try
{
    Log.Information("Starting ISO Document Management System v3...");

    var builder = WebApplication.CreateBuilder(args);

    // ── Serilog (replace default logging) ─────────────────────────────────────
    builder.Host.UseSerilog((ctx, services, config) =>
        config.ReadFrom.Configuration(ctx.Configuration)
              .ReadFrom.Services(services)
              .Enrich.FromLogContext());

    // ── Services ──────────────────────────────────────────────────────────────
    builder.Services
        .AddApplicationServices()                                   // MediatR + Validators + AutoMapper
        .AddInfrastructureServices(builder.Configuration)           // EF Core + Blob + ES + Redis + JWT
        .AddWebApiServices(builder.Configuration);                  // Controllers + Swagger + Auth + CORS

    // Suppress default model state validation (we use FluentValidation via MediatR pipeline)
    builder.Services.Configure<ApiBehaviorOptions>(options =>
        options.SuppressModelStateInvalidFilter = true);

    var app = builder.Build();

    // ── HTTP Pipeline ─────────────────────────────────────────────────────────
    // Order is critical — exception handler must be first to catch everything
    app.UseExceptionHandling();             // Custom middleware → RFC 7807 ProblemDetails

    app.UseSerilogRequestLogging(opts =>
    {
        opts.MessageTemplate =
            "HTTP {RequestMethod} {RequestPath} responded {StatusCode} in {Elapsed:0.0000}ms";
    });

    if (app.Environment.IsDevelopment())
    {
        app.UseSwagger();
        app.UseSwaggerUI(c =>
        {
            c.SwaggerEndpoint("/swagger/v1/swagger.json", "ISO DMS API v1");
            c.RoutePrefix  = "swagger";
            c.DocExpansion = Swashbuckle.AspNetCore.SwaggerUI.DocExpansion.None;
        });
    }

    app.UseHttpsRedirection();
    app.UseCors("DefaultCors");
    app.UseAuthentication();
    app.UseAuthorization();

    app.MapControllers();
    app.MapHealthChecks("/health");

    // ── Apply EF Core migrations on startup (dev only) ────────────────────────
    if (app.Environment.IsDevelopment())
        await IsoDoc.Infrastructure.DependencyInjection.ApplyMigrationsAsync(app.Services);

    await app.RunAsync();
}
catch (Exception ex) when (ex is not HostAbortedException)
{
    Log.Fatal(ex, "Application terminated unexpectedly.");
}
finally
{
    Log.CloseAndFlush();
}

 
3. WebApiServiceExtensions — DI Registration
AddWebApiServices() đăng ký toàn bộ WebAPI concerns trong một method. Bao gồm Controllers, Swagger với Bearer auth, JWT middleware, Authorization policies, CORS, và Health Checks.

3.1  Authorization Policies
Policy name	Roles được phép
RequireController	DocumentController, ISOManager, SystemAdmin
RequireApprover	QAOfficer, SafetyOfficer, ISMSOfficer, ISOManager
RequireISOManager	ISOManager, SystemAdmin
RequireSystemAdmin	SystemAdmin
RequireViewer	Tất cả authenticated users (default)

using IsoDoc.Infrastructure.Identity;
using Microsoft.AspNetCore.Authentication.JwtBearer;
using Microsoft.IdentityModel.Tokens;
using Microsoft.OpenApi.Models;
using System.Text;

namespace IsoDoc.WebAPI.Extensions;

public static class WebApiServiceExtensions
{
    public static IServiceCollection AddWebApiServices(
        this IServiceCollection services,
        IConfiguration          configuration)
    {
        // ── Controllers ───────────────────────────────────────────────────
        services.AddControllers(opts =>
        {
            // Return 406 for unsupported Accept headers
            opts.ReturnHttpNotAcceptable = true;
        })
        .AddJsonOptions(opts =>
        {
            // Enums as strings in JSON (e.g. "Published" not 4)
            opts.JsonSerializerOptions.Converters.Add(
                new System.Text.Json.Serialization.JsonStringEnumConverter());
            opts.JsonSerializerOptions.DefaultIgnoreCondition =
                System.Text.Json.Serialization.JsonIgnoreCondition.WhenWritingNull;
        });

        // ── Swagger / OpenAPI ─────────────────────────────────────────────
        services.AddEndpointsApiExplorer();
        services.AddSwaggerGen(c =>
        {
            c.SwaggerDoc("v1", new OpenApiInfo
            {
                Title       = "ISO Document Management System API",
                Version     = "v1",
                Description = "API quản lý tài liệu theo tiêu chuẩn ISO 9001, 45001, 27001.",
                Contact     = new OpenApiContact { Name = "ISO DMS Team", Email = "dev@isodms.internal" }
            });

            // Add JWT Bearer auth to Swagger UI
            var jwtScheme = new OpenApiSecurityScheme
            {
                Name         = "Authorization",
                Type         = SecuritySchemeType.Http,
                Scheme       = "bearer",
                BearerFormat = "JWT",
                In           = ParameterLocation.Header,
                Description  = "Nhập JWT token. Ví dụ: Bearer eyJhbGc..."
            };
            c.AddSecurityDefinition("Bearer", jwtScheme);
            c.AddSecurityRequirement(new OpenApiSecurityRequirement
            {
                {
                    new OpenApiSecurityScheme
                    {
                        Reference = new OpenApiReference
                            { Type = ReferenceType.SecurityScheme, Id = "Bearer" }
                    },
                    Array.Empty<string>()
                }
            });

            // Include XML comments from controllers
            var xmlFile = $"{System.Reflection.Assembly.GetExecutingAssembly().GetName().Name}.xml";
            var xmlPath = Path.Combine(AppContext.BaseDirectory, xmlFile);
            if (File.Exists(xmlPath)) c.IncludeXmlComments(xmlPath);
        });

        // ── JWT Authentication ────────────────────────────────────────────
        var jwtOpts = configuration.GetSection(JwtOptions.Section).Get<JwtOptions>()!;
        var key     = Encoding.UTF8.GetBytes(jwtOpts.SecretKey);

        services
            .AddAuthentication(JwtBearerDefaults.AuthenticationScheme)
            .AddJwtBearer(opts =>
            {
                opts.TokenValidationParameters = new TokenValidationParameters
                {
                    ValidateIssuer           = true,
                    ValidIssuer              = jwtOpts.Issuer,
                    ValidateAudience         = true,
                    ValidAudience            = jwtOpts.Audience,
                    ValidateIssuerSigningKey = true,
                    IssuerSigningKey         = new SymmetricSecurityKey(key),
                    ValidateLifetime         = true,
                    ClockSkew                = TimeSpan.Zero,      // No grace period
                };

                // Support Blazor Server — token from query string for SignalR
                opts.Events = new JwtBearerEvents
                {
                    OnMessageReceived = ctx =>
                    {
                        var token = ctx.Request.Query["access_token"];
                        var path  = ctx.HttpContext.Request.Path;
                        if (!string.IsNullOrEmpty(token) && path.StartsWithSegments("/hubs"))
                            ctx.Token = token;
                        return Task.CompletedTask;
                    }
                };
            });

        // ── Authorization Policies ────────────────────────────────────────
        services.AddAuthorizationBuilder()
            .AddPolicy("RequireController",   p => p.RequireRole("DocumentController", "ISOManager", "SystemAdmin"))
            .AddPolicy("RequireApprover",     p => p.RequireRole("QAOfficer", "SafetyOfficer", "ISMSOfficer", "ISOManager"))
            .AddPolicy("RequireISOManager",   p => p.RequireRole("ISOManager", "SystemAdmin"))
            .AddPolicy("RequireSystemAdmin",  p => p.RequireRole("SystemAdmin"))
            .AddPolicy("RequireViewer",       p => p.RequireAuthenticatedUser());

        // ── CORS ──────────────────────────────────────────────────────────
        var allowedOrigins = configuration
            .GetSection("Cors:AllowedOrigins")
            .Get<string[]>() ?? new[] { "http://localhost:5000", "https://localhost:5001" };

        services.AddCors(opts =>
            opts.AddPolicy("DefaultCors", policy =>
                policy.WithOrigins(allowedOrigins)
                      .AllowAnyMethod()
                      .AllowAnyHeader()
                      .AllowCredentials()));

        // ── Health Checks ─────────────────────────────────────────────────
        services.AddHealthChecks()
            .AddDbContextCheck<IsoDoc.Infrastructure.Persistence.AppDbContext>("database")
            .AddRedis(configuration.GetConnectionString("Redis") ?? "localhost:6379", "redis");

        return services;
    }

    public static IApplicationBuilder UseExceptionHandling(this IApplicationBuilder app)
        => app.UseMiddleware<ExceptionHandlingMiddleware>();
}

 
4. ExceptionHandlingMiddleware — RFC 7807
Middleware bắt mọi exception chưa được xử lý và map sang ProblemDetails JSON chuẩn RFC 7807. Client luôn nhận cùng một format, không bao giờ thấy stack trace trong production.

4.1  Exception → HTTP status mapping
Exception	HTTP	Khi nào xảy ra
ValidationException (FluentValidation)	400	Validator từ chối input — có danh sách lỗi từng field
DocumentNotFoundException	404	Document ID không tồn tại trong DB
DomainException	422	Business rule bị vi phạm
InvalidDocumentWorkflowStateException	422	Sai state transition (e.g. publish Draft)
UnauthorizedAccessException	401	Chưa đăng nhập hoặc token hết hạn
ForbiddenAccessException	403	Thiếu RBAC permission cụ thể
OperationCanceledException	499	Client ngắt kết nối — không trả response
Tất cả exception khác	500	Lỗi hệ thống — stack trace ẩn trong production

500 errors: log đầy đủ với _logger.LogError(ex, ...). 4xx errors: chỉ log Warning — không cần stack trace vì là lỗi client.

using FluentValidation;
using IsoDoc.Application.Common.Behaviours;
using IsoDoc.Domain.Exceptions;
using Microsoft.AspNetCore.Mvc;
using System.Text.Json;

namespace IsoDoc.WebAPI.Middleware;

/// <summary>
/// Global exception handler middleware.
/// Catches ALL unhandled exceptions and maps them to RFC 7807 ProblemDetails JSON.
/// Ensures no raw stack traces are ever leaked to clients.
///
/// Mapping table:
///   ValidationException          → 400 Bad Request
///   DomainException              → 422 Unprocessable Entity
///   DocumentNotFoundException    → 404 Not Found
///   InvalidWorkflowStateException→ 422 Unprocessable Entity
///   UnauthorizedAccessException  → 401 Unauthorized
///   ForbiddenAccessException     → 403 Forbidden
///   OperationCanceledException   → 499 Client Closed Request (no body)
///   Everything else              → 500 Internal Server Error
/// </summary>
public sealed class ExceptionHandlingMiddleware
{
    private readonly RequestDelegate                       _next;
    private readonly ILogger<ExceptionHandlingMiddleware>  _logger;
    private readonly IHostEnvironment                      _env;

    private static readonly JsonSerializerOptions _jsonOpts = new()
    {
        PropertyNamingPolicy = JsonNamingPolicy.CamelCase
    };

    public ExceptionHandlingMiddleware(
        RequestDelegate                      next,
        ILogger<ExceptionHandlingMiddleware> logger,
        IHostEnvironment                     env)
    {
        _next   = next;
        _logger = logger;
        _env    = env;
    }

    public async Task InvokeAsync(HttpContext ctx)
    {
        try
        {
            await _next(ctx);
        }
        catch (OperationCanceledException) when (ctx.RequestAborted.IsCancellationRequested)
        {
            // Client disconnected — no response needed
            ctx.Response.StatusCode = 499;
        }
        catch (Exception ex)
        {
            await HandleExceptionAsync(ctx, ex);
        }
    }

    private async Task HandleExceptionAsync(HttpContext ctx, Exception ex)
    {
        var (statusCode, title, detail, errors) = ex switch
        {
            // FluentValidation — returns all field errors
            ValidationException ve => (
                StatusCodes.Status400BadRequest,
                "Dữ liệu không hợp lệ",
                "Một hoặc nhiều trường dữ liệu không đúng định dạng.",
                ve.Errors
                  .GroupBy(e => e.PropertyName)
                  .ToDictionary(
                      g => g.Key,
                      g => g.Select(e => e.ErrorMessage).ToArray())),

            // Domain business rule violation
            DocumentNotFoundException =>
                (StatusCodes.Status404NotFound,
                 "Không tìm thấy tài liệu",
                 ex.Message,
                 (Dictionary<string, string[]>?)null),

            InvalidDocumentWorkflowStateException =>
                (StatusCodes.Status422UnprocessableEntity,
                 "Trạng thái tài liệu không hợp lệ",
                 ex.Message,
                 (Dictionary<string, string[]>?)null),

            DomainException =>
                (StatusCodes.Status422UnprocessableEntity,
                 "Vi phạm quy tắc nghiệp vụ",
                 ex.Message,
                 (Dictionary<string, string[]>?)null),

            // Auth
            UnauthorizedAccessException =>
                (StatusCodes.Status401Unauthorized,
                 "Chưa xác thực",
                 "Bạn cần đăng nhập để thực hiện thao tác này.",
                 (Dictionary<string, string[]>?)null),

            ForbiddenAccessException fax =>
                (StatusCodes.Status403Forbidden,
                 "Không có quyền truy cập",
                 $"Bạn không có quyền '{fax.Permission}'.",
                 (Dictionary<string, string[]>?)null),

            // Catch-all
            _ => (StatusCodes.Status500InternalServerError,
                  "Lỗi hệ thống",
                  _env.IsDevelopment()
                      ? ex.ToString()
                      : "Đã xảy ra lỗi không mong muốn. Vui lòng thử lại.",
                  (Dictionary<string, string[]>?)null)
        };

        // Log server errors; not client errors (400/422/404)
        if (statusCode >= 500)
            _logger.LogError(ex, "Unhandled exception: {Message}", ex.Message);
        else
            _logger.LogWarning("Client error {StatusCode}: {Message}", statusCode, ex.Message);

        var problem = new ValidationProblemDetails
        {
            Type     = $"https://isodms.internal/errors/{statusCode}",
            Title    = title,
            Status   = statusCode,
            Detail   = detail,
            Instance = ctx.Request.Path,
            Extensions = { ["traceId"] = ctx.TraceIdentifier }
        };

        if (errors is not null)
            foreach (var kv in errors)
                problem.Errors[kv.Key] = kv.Value;

        ctx.Response.StatusCode  = statusCode;
        ctx.Response.ContentType = "application/problem+json";

        await ctx.Response.WriteAsync(
            JsonSerializer.Serialize(problem, _jsonOpts));
    }
}

 
5. ApiResponse<T> + ApiControllerBase
Tất cả responses bọc trong ApiResponse<T> để client có cấu trúc nhất quán. ApiControllerBase cung cấp helper methods: OkResult(), PagedResult(), CreatedResult(), FromResult() để giảm boilerplate trong controllers.

Mediator được lấy từ HttpContext.RequestServices thay vì constructor inject — giảm constructor parameter cho controllers khi có nhiều endpoints.

using IsoDoc.Application.Common.Models;
using MediatR;
using Microsoft.AspNetCore.Mvc;

namespace IsoDoc.WebAPI.Models;

/// <summary>
/// Standardised JSON envelope for all API responses.
/// Queries return { success, data, pagination? }
/// Commands return { success, data? }
/// Errors are handled by ExceptionHandlingMiddleware → RFC 7807 ProblemDetails.
/// </summary>
public sealed class ApiResponse<T>
{
    public bool T      IsSuccess { get; init; }
    public T?   Data      { get; init; }
    public PaginationMeta? Pagination { get; init; }

    public static ApiResponse<T> Ok(T data, PaginationMeta? pagination = null)
        => new() { IsSuccess = true, Data = data, Pagination = pagination };
}

public sealed class PaginationMeta
{
    public int Page       { get; init; }
    public int PageSize   { get; init; }
    public int TotalCount { get; init; }
    public int TotalPages { get; init; }
    public bool HasNext   { get; init; }
    public bool HasPrev   { get; init; }

    public static PaginationMeta From<T>(PagedList<T> paged)
        => new()
        {
            Page       = paged.Page,
            PageSize   = paged.PageSize,
            TotalCount = paged.TotalCount,
            TotalPages = paged.TotalPages,
            HasNext    = paged.HasNextPage,
            HasPrev    = paged.HasPreviousPage,
        };
}

namespace IsoDoc.WebAPI.Controllers;

/// <summary>
/// Base controller with MediatR + helper methods for consistent responses.
/// All controllers inherit this.
/// </summary>
[ApiController]
[Produces("application/json")]
public abstract class ApiControllerBase : ControllerBase
{
    private ISender? _mediator;

    protected ISender Mediator =>
        _mediator ??= HttpContext.RequestServices.GetRequiredService<ISender>();

    protected IActionResult OkResult<T>(T data) =>
        Ok(ApiResponse<T>.Ok(data));

    protected IActionResult PagedResult<T>(PagedList<T> paged) =>
        Ok(ApiResponse<IReadOnlyList<T>>.Ok(
            paged.Items,
            PaginationMeta.From(paged)));

    protected IActionResult CreatedResult<T>(T data, string routeName, object routeValues) =>
        CreatedAtRoute(routeName, routeValues,
            ApiResponse<T>.Ok(data));

    /// <summary>
    /// Converts a Result to IActionResult.
    /// Result.Failure triggers ExceptionHandlingMiddleware via throw,
    /// or returns 422 inline — depending on caller preference.
    /// </summary>
    protected IActionResult FromResult<T>(Result<T> result)
    {
        if (result.IsSuccess)
            return Ok(ApiResponse<T>.Ok(result.Value!));

        return result.ErrorCode switch
        {
            "DOCUMENT_NOT_FOUND"      => NotFound(Problem(result.Error)),
            "DOCUMENT_CODE_DUPLICATE" => Conflict(Problem(result.Error)),
            "UNAUTHORIZED"            => Forbid(),
            _                         => UnprocessableEntity(Problem(result.Error))
        };
    }

    protected IActionResult FromResult(Result result)
    {
        if (result.IsSuccess) return NoContent();
        return UnprocessableEntity(Problem(result.Error));
    }

    private static ProblemDetails Problem(string? detail) => new()
    {
        Detail = detail,
        Status = StatusCodes.Status422UnprocessableEntity
    };
}

using IsoDoc.Application.Documents.Commands.SubmitForApproval;
using IsoDoc.Application.Documents.Commands.UploadDocument;
using IsoDoc.Application.Documents.Queries;
using IsoDoc.Domain.Enums;
using Microsoft.AspNetCore.Authorization;
using Microsoft.AspNetCore.Mvc;

namespace IsoDoc.WebAPI.Controllers;

/// <summary>
/// Documents resource controller.
/// All endpoints require authentication. Write endpoints require specific roles.
/// </summary>
[Route("api/v1/[controller]")]
[Authorize]
public sealed class DocumentsController : ApiControllerBase
{
    // ── GET /api/v1/documents ─────────────────────────────────────────────────

    /// <summary>Tìm kiếm và lọc danh sách tài liệu.</summary>
    /// <param name="keyword">Từ khóa tìm kiếm (full-text)</param>
    /// <param name="standard">Lọc theo tiêu chuẩn ISO (ISO9001, ISO45001, ISO27001)</param>
    /// <param name="status">Lọc theo trạng thái tài liệu</param>
    /// <param name="category">Lọc theo phân loại</param>
    /// <param name="fromDate">Từ ngày cập nhật</param>
    /// <param name="toDate">Đến ngày cập nhật</param>
    /// <param name="page">Trang (bắt đầu từ 1)</param>
    /// <param name="pageSize">Số item mỗi trang (1-100, mặc định 20)</param>
    /// <param name="sortBy">Sắp xếp theo trường (UpdatedAt, Title, DocumentCode, Status)</param>
    /// <param name="sortDesc">Sắp xếp giảm dần</param>
    [HttpGet]
    [ProducesResponseType(StatusCodes.Status200OK)]
    public async Task<IActionResult> Search(
        [FromQuery] string?           keyword   = null,
        [FromQuery] IsoStandard?      standard  = null,
        [FromQuery] DocumentStatus?   status    = null,
        [FromQuery] DocumentCategory? category  = null,
        [FromQuery] DateTime?         fromDate  = null,
        [FromQuery] DateTime?         toDate    = null,
        [FromQuery] int               page      = 1,
        [FromQuery] int               pageSize  = 20,
        [FromQuery] string            sortBy    = "UpdatedAt",
        [FromQuery] bool              sortDesc  = true,
        CancellationToken             ct        = default)
    {
        var query = new SearchDocumentsQuery
        {
            Keyword  = keyword,
            Standard = standard,
            Status   = status,
            Category = category,
            FromDate = fromDate,
            ToDate   = toDate,
            Page     = page,
            PageSize = pageSize,
            SortBy   = sortBy,
            SortDesc = sortDesc,
        };

        var result = await Mediator.Send(query, ct);
        return result.IsSuccess
            ? PagedResult(result.Value!)
            : FromResult(result);
    }

    // ── GET /api/v1/documents/{id} ────────────────────────────────────────────

    /// <summary>Lấy chi tiết tài liệu theo ID.</summary>
    [HttpGet("{id:guid}", Name = "GetDocumentById")]
    [ProducesResponseType(StatusCodes.Status200OK)]
    [ProducesResponseType(StatusCodes.Status404NotFound)]
    public async Task<IActionResult> GetById(
        Guid              id,
        [FromQuery] bool  includeVersions = true,
        CancellationToken ct              = default)
    {
        var result = await Mediator.Send(
            new GetDocumentByIdQuery(id, includeVersions), ct);

        return FromResult(result);
    }

    // ── POST /api/v1/documents ────────────────────────────────────────────────

    /// <summary>Tải lên tài liệu mới (Draft). Yêu cầu role: DocumentController hoặc ISOManager.</summary>
    [HttpPost]
    [Authorize(Policy = "RequireController")]
    [ProducesResponseType(StatusCodes.Status201Created)]
    [ProducesResponseType(StatusCodes.Status400BadRequest)]
    [ProducesResponseType(StatusCodes.Status409Conflict)]
    [Consumes("multipart/form-data")]
    public async Task<IActionResult> Upload(
        [FromForm] UploadDocumentRequest request,
        CancellationToken                ct = default)
    {
        if (request.File is null || request.File.Length == 0)
            return BadRequest(Problem("File không được rỗng."));

        // Compute SHA-256 checksum on server side for integrity
        string checksumHex;
        using (var ms = new MemoryStream())
        {
            await request.File.CopyToAsync(ms, ct);
            checksumHex = Convert.ToHexString(
                System.Security.Cryptography.SHA256.HashData(ms.ToArray()))
                .ToLowerInvariant();
        }

        var command = new UploadDocumentCommand
        {
            Title        = request.Title,
            DocumentCode = request.DocumentCode,
            Standard     = request.Standard,
            Category     = request.Category,
            Description  = request.Description,
            Tags         = request.Tags ?? new List<string>(),
            FileStream   = request.File.OpenReadStream(),
            FileName     = request.File.FileName,
            ContentType  = request.File.ContentType,
            FileSize     = request.File.Length,
            ChecksumHex  = checksumHex,
            ChangeNote   = request.ChangeNote,
        };

        var result = await Mediator.Send(command, ct);

        return result.IsSuccess
            ? CreatedResult(result.Value, "GetDocumentById", new { id = result.Value })
            : FromResult(result);
    }

/// <summary>Authentication — login, refresh, logout.</summary>
[Route("api/v1/[controller]")]
public sealed class AuthController : ApiControllerBase
{
    private readonly JwtTokenService _jwt;

    public AuthController(JwtTokenService jwt) => _jwt = jwt;

    /// <summary>
    /// Đăng nhập bằng email + password.
    /// Trả về JWT access token và refresh token.
    /// </summary>
    [HttpPost("login")]
    [AllowAnonymous]
    [ProducesResponseType(StatusCodes.Status200OK)]
    [ProducesResponseType(StatusCodes.Status401Unauthorized)]
    public async Task<IActionResult> Login(
        [FromBody] LoginRequest request,
        CancellationToken       ct = default)
    {
        // Production: dispatch LoginCommand → validate credentials via ASP.NET Identity
        // → generate JWT via JwtTokenService → store refresh token in DB
        await Task.CompletedTask;

        // Demo response
        var accessToken = _jwt.GenerateAccessToken(
            Guid.NewGuid(), request.Email, "Demo User",
            new[] { "DocumentController" });

        return OkResult(new LoginResponse
        {
            AccessToken  = accessToken,
            RefreshToken = Guid.NewGuid().ToString("N"),
            ExpiresIn    = 3600,
            TokenType    = "Bearer"
        });
    }

    /// <summary>Làm mới access token bằng refresh token.</summary>
    [HttpPost("refresh")]
    [AllowAnonymous]
    [ProducesResponseType(StatusCodes.Status200OK)]
    [ProducesResponseType(StatusCodes.Status401Unauthorized)]
    public async Task<IActionResult> Refresh(
        [FromBody] RefreshRequest request,
        CancellationToken         ct = default)
    {
        // Production: validate refresh token from DB → issue new access + refresh token
        await Task.CompletedTask;
        return Unauthorized();
    }

    [Authorize]

    }
